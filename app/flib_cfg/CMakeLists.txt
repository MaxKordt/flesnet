add_executable(flib_cfg flib_cfg.cpp)

target_compile_definitions(flib_cfg PUBLIC BOOST_ALL_DYN_LINK)

  ${Boost_LIBRARIES} ${CMAKE_THREAD_LIBS_INIT} rt
)
